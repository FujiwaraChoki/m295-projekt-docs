    <h2>{group.name}</h2>

      <div key={endpoint.name}>
        <h3>{endpoint.name}</h3>
        <p>{endpoint.description}</p>
        <h4>Beispiel</h4>

        <h4>Parameter</h4>
        <table>

              </tr>
            ))}
          </tbody>
        </table>
        <h4>Antwort</h4>
        <table>
          <thead>
            <tr>
              <th>Name</th>

          <tbody>

              <tr key={response.name}>
                <td>{response.name}</td>
                <td>{response.type}</td>
                <td>{response.description}</td>
              </tr>
            ))}
          </tbody>

    ))}
  </div>
